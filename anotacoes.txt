variável - possui um espaço reservado na memória para armazenar informação
constante - valor que não muda

let
    não é mais var, usa let
    só existe dentro da função
    var é varíavel global
    let é varíavel local

   -tipado: precisa dizer qual o tipo de variável

const
    quando a variável é constante
    desde que é criada, o valor atribúido não pode ser alterado
    deve ser definida desde o começo

camelcase
    exemplo nome variável: diaDaSemana
    let diaDaSemana = "domingo"

atribuir valor a variáveis =
    let nome = "alessandra"

comparar valor de variáveis ==
    "sabado" == "domingo"
    deve retornar false

    14 == "14"
    deve retornar true

comparar valor de variáveis levando em consideração o tipo de dado ===
    14 === "14"
    deve retornar false

condicional E: usa &&

condicional OU: usa ||
